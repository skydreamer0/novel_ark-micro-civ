## 前置準備

### Step 1: 建立批次腳本

```powershell
# 統計各章中文字數的腳本
$root = "C:\Users\User\OneDrive - MSFT (1)\0.專案\novel\novel_ark-micro-civ\01_NOVEL_CONTENT"
$vols = @("01_第一卷_艙底種火", "02_第二卷_裂縫稅戰", "03_第三卷_聖城試煉", "04_第四卷_投放名冊")
$results = @()
foreach ($vol in $vols) {
    $path = Join-Path $root $vol
    if (Test-Path $path) {
        Get-ChildItem $path -Filter "*.md" | ForEach-Object {
            $content = Get-Content -Raw $_.FullName
            # 只算中文字元（去除Markdown語法、空白、標題）
            $body = $content -replace '^# .+$', '' -replace '^\s*$', ''
            $chineseChars = [regex]::Matches($body, '[\u4e00-\u9fff]').Count
            $results += [PSCustomObject]@{
                File = $_.Name
                Path = $_.FullName
                ChineseChars = $chineseChars
                NeedExpansion = $chineseChars -lt 3000
                Shortfall = [Math]::Max(0, 3000 - $chineseChars)
            }
        }
    }
}
$results | Export-Csv -Path "wordcount_audit.csv" -Encoding UTF8
```
